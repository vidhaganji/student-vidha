        import seaborn as sns
import matplotlib.pyplot as plt

# Load the flights dataset (example data)
flights_data = sns.load_dataset("flights")

# Pivot the data to have months as rows, years as columns, and number of passengers as values
flights_pivot = flights_data.pivot("month", "year", "passengers")

# Create a heatmap using Seaborn
plt.figure(figsize=(12, 8))
sns.heatmap(flights_pivot, annot=True, fmt="d", cmap="YlGnBu")

# Set the title and labels
plt.title("Passenger Counts by Month and Year")
plt.xlabel("Year")
plt.ylabel("Month")

# Show the plot
plt.show()
